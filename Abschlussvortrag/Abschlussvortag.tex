\begin{frame}
\frametitle {Charakteristika der Peaks}

Peak charakterisiert durch:
\begin{itemize}
 \item<+-> Lage des Maximums
 \item<+-> Breite
 \begin{itemize}
  \item Interquartilskoeffizient
  \item $IQR = Q_{75} - Q_{25}$
 \end{itemize}
 \item<+-> Form
  \begin{itemize}
    \item Idealfall: Gaußkurve 
    \item Abweichung: Fronting, Tailing %(Bild an Tafel mit Tailing)
    \item Quartilskoeffizient
    \begin{equation*}
    \frac{Q_{\alpha} + Q_{1-\alpha} -2\cdot Q_{50} }{ Q_{1-\alpha} - Q_{\alpha}}
    \end{equation*}
    $ > 0$ : rechtsschief\\
    $ < 0$ : linksschief
  \end{itemize}
\end{itemize}

\end {frame}

\section{2-Zustände Modell}
\subsection*{Annahmen}

\begin{frame}
\frametitle {Modell für die Chromatographie}
 
\onslide<1->{Prinzip:}
\begin{itemize}
 \item <2-> 2 Phasen: stationär und mobil
 \item <3-> Wechsel dazwischen, bzw. Verweilen in der Phase
\end{itemize}

\onslide<1->{Modell:}
\begin{itemize}
 \item <2-> 2 Zustände: $s$ und $m$
 \item <3-> Wechselwahrscheinlichkeiten
 \begin{itemize}
  \item $s \rightarrow s: p_s$
  \item $s \rightarrow m: 1-p_s$
  \item $m \rightarrow m: p_m$
  \item $m \rightarrow s: 1-p_m$ 
 \end{itemize}

\end{itemize}

 
\end{frame}

 \frametitle{Simulationsarten}
 
 \begin{itemize}
  \item<+-> Teilchensimulation: \\
  Simuliere $n$ Teilchen,\\ für Säule der Länge $\ell$,\\ für maximal $t_{\text{max}}$ Schritte,\\ verwalte Orte und Zustände der Teilchen
  \begin{itemize}
   \item Step-by-Step
   \item By-Event
  \end{itemize}
  \item<+-> PAA: \\
  Berechne Verteilung der Wartezeit für Wert $\ell$\\ maximale Wartezeit von $t_{\text{max}}$ 
 \end{itemize}
\end{frame}
